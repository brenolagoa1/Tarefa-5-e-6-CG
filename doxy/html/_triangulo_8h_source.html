33 // Vetor que representa a primeira aresta do triângulo
34 vec3 edge1 = v1 - v0;
35 // Vetor que representa a segunda aresta do triângulo
36 vec3 edge2 = v2 - v0;
37 // Vetor perpendicular ao plano do triângulo e ao raio
38 vec3 h = cross(r.direction(), edge2);
39 // Produto escalar entre a primeira aresta e o vetor h
40 double a = dot(edge1, h);
41
42 // Verifica se o raio é paralelo ou quase paralelo ao triângulo
43 if (a > -1e-6 && a < 1e-6) {
44 return false;
45 }
46
47 // Fator de escala para calcular a interseção
48 double f = 1.0 / a;
49 // Vetor da origem do raio ao primeiro vértice do triângulo
50 vec3 s = r.origin() - v0;
51 // Coordenada baricêntrica u
52 double u = f * dot(s, h);
53
54 // Verifica se o ponto de interseção está fora do triângulo na direção da primeira aresta
55 if (u < 0.0 || u > 1.0) {
56 return false;
57 }
58
59 // Vetor perpendicular ao vetor s e à primeira aresta
60 vec3 q = cross(s, edge1);
61 // Coordenada baricêntrica v
62 double v = f * dot(r.direction(), q);
63
64 // Verifica se o ponto de interseção está fora do triângulo na direção da segunda aresta
65 if (v < 0.0 || u + v > 1.0) {
66 return false;
67 }
68

70 double t = f * dot(edge2, q);
71 // Verifica se a interseção está dentro do intervalo de tempo do raio e t é positivo

93 vec3 toBarycentricCoordinates(const vec3& point) const {
94 // Vetor da origem do triângulo ao ponto
95 vec3 v2 = point - v0;
96
97 // Produtos escalares entre as arestas do triângulo
98 double d00 = dot(v1 - v0, v1 - v0);
99 double d01 = dot(v1 - v0, v2);
100 double d11 = dot(v2, v2);
101 double d20 = dot(v2, v1 - v0);
102 double d21 = dot(v2, v2 - v0);
103
104 // Calcula as coordenadas baricêntricas u, v e w
105 double denominator = d00 * d11 - d01 * d01;
106 double v = (d11 * d20 - d01 * d21) / denominator;
107 double w = (d00 * d21 - d01 * d20) / denominator;
108 double u = 1.0 - v - w;
109
110 return vec3(u, v, w);
111 }
